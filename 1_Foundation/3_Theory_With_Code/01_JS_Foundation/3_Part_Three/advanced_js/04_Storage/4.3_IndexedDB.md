# IndexedDB Summary

IndexedDB is a powerful browser-based database for offline apps, supporting transactions, key ranges, indexes, and much larger storage than localStorage.

## Key Features

- **Large storage** - megabytes to gigabytes (browser dependent)
- **Transactions** - atomic operations (all succeed or all fail)
- **Key ranges** - query by key or key range
- **Indexes** - search by any object field
- **Stores any data** - objects, arrays, primitives (via structured cloning)

## Opening a Database

```javascript
let openRequest = indexedDB.open("database", 1); // name, version

openRequest.onupgradeneeded = function (event) {
  // Called when:
  // - Database doesn't exist (version 0)
  // - Version specified is higher than existing
  let db = openRequest.result;

  // Create object store here
  let store = db.createObjectStore("books", { keyPath: "id" });
  store.createIndex("price_idx", "price");
};

openRequest.onsuccess = function () {
  let db = openRequest.result;
  console.log("Database ready");

  // Handle parallel version updates
  db.onversionchange = function () {
    db.close();
    alert("Database updated, please reload");
  };
};

openRequest.onerror = function () {
  console.error("Error opening DB:", openRequest.error);
};

openRequest.onblocked = function () {
  // Another tab has outdated DB open
  alert("Please close other tabs with this site");
};
```

## Object Store

Storage container (like a table):

```javascript
// Create store (in upgradeneeded only)
let store = db.createObjectStore("books", { keyPath: "id" });
// or auto-increment key
let store = db.createObjectStore("books", { autoIncrement: true });
// or no key path (provide key manually on add)
let store = db.createObjectStore("books");

// Delete store
db.deleteObjectStore("books");
```

## Transactions

All data operations must be in a transaction:

```javascript
let transaction = db.transaction("books", "readonly"); // or 'readwrite'
let store = transaction.objectStore("books");

// Perform operations
let request = store.get("js");
request.onsuccess = () => console.log(request.result);

// Transaction events
transaction.oncomplete = () => console.log("Transaction complete");
transaction.onabort = () => console.log("Transaction aborted");
```

## CRUD Operations

```javascript
let transaction = db.transaction("books", "readwrite");
let store = transaction.objectStore("books");

// Add (fails if key exists)
let addRequest = store.add({ id: "js", price: 10, title: "JavaScript Guide" });

// Put (overwrites if exists)
let putRequest = store.put({ id: "js", price: 15, title: "JavaScript Guide" });

// Get by key
let getRequest = store.get("js");

// Delete by key
let deleteRequest = store.delete("js");

// Clear all
let clearRequest = store.clear();

// Get all
let getAllRequest = store.getAll();

// Count
let countRequest = store.count();
```

## Key Ranges

```javascript
// Range types
IDBKeyRange.lowerBound(10); // ≥10
IDBKeyRange.lowerBound(10, true); // >10
IDBKeyRange.upperBound(20); // ≤20
IDBKeyRange.upperBound(20, true); // <20
IDBKeyRange.bound(10, 20); // 10 ≤ x ≤ 20
IDBKeyRange.bound(10, 20, true, true); // 10 < x < 20
IDBKeyRange.only("js"); // exact match

// Usage
let range = IDBKeyRange.bound("css", "html");
let request = store.getAll(range);
```

## Indexes

Search by non-key fields:

```javascript
// Create index (in upgradeneeded)
let store = db.createObjectStore("books", { keyPath: "id" });
let priceIndex = store.createIndex("price_idx", "price", {
  unique: false, // Allow duplicate values
  multiEntry: false, // For array values
});

// Use index for search
let transaction = db.transaction("books");
let store = transaction.objectStore("books");
let index = store.index("price_idx");

// Search by exact price
let request = index.getAll(10);

// Search price range
let cheapBooks = index.getAll(IDBKeyRange.upperBound(5));
```

## Cursors (Memory-efficient iteration)

```javascript
let request = store.openCursor();

request.onsuccess = function () {
  let cursor = request.result;
  if (cursor) {
    console.log("Key:", cursor.key);
    console.log("Value:", cursor.value);

    cursor.continue(); // Move to next
    // cursor.advance(2); // Skip 2 items
  } else {
    console.log("No more items");
  }
};

// Cursor over index
let request = priceIndex.openCursor(IDBKeyRange.upperBound(5));

// Get keys only
let keyRequest = store.openKeyCursor();
```

## Promise Wrapper (idb)

Install: `npm install idb` or use CDN

```javascript
import { openDB } from "idb";

// Open database
let db = await openDB("store", 1, {
  upgrade(db, oldVersion, newVersion, transaction) {
    let store = db.createObjectStore("books", { keyPath: "id" });
    store.createIndex("price_idx", "price");
  },
});

// Add data
let tx = db.transaction("books", "readwrite");
let store = tx.objectStore("books");
await store.add({ id: "js", price: 10 });
await store.add({ id: "css", price: 8 });
await tx.done; // Wait for completion

// Query
let book = await db.get("books", "js");
let books = await db.getAll("books");
let cheap = await db.getAllFromIndex(
  "books",
  "price_idx",
  IDBKeyRange.upperBound(5),
);

// Delete
await db.delete("books", "js");
await db.clear("books");
```

## Error Handling

```javascript
try {
  await db.add("books", { id: "js", price: 10 });
} catch (err) {
  if (err.name === "ConstraintError") {
    console.log("Book already exists");
  } else {
    throw err;
  }
}

// Catch unhandled rejections
window.addEventListener("unhandledrejection", (event) => {
  console.error("Unhandled error:", event.reason);
});
```

## Transaction Auto-Commit Pitfall

Transactions auto-commit when code finishes. Async operations break them:

```javascript
// ❌ Wrong - fetch will cause transaction to close
let tx = db.transaction("books", "readwrite");
await tx.store.add(book);
await fetch("/api"); // Transaction closes here
await tx.store.add(book2); // Error: TransactionInactiveError

// ✅ Correct - prepare data first
let book1 = { id: "js", price: 10 };
let book2 = { id: "css", price: 8 };
let response = await fetch("/api");
let tx = db.transaction("books", "readwrite");
await tx.store.add(book1);
await tx.store.add(book2);
await tx.done;
```

## Methods Quick Reference

| Operation     | Method                                      |
| ------------- | ------------------------------------------- |
| Open DB       | `indexedDB.open(name, version)`             |
| Delete DB     | `indexedDB.deleteDatabase(name)`            |
| Create store  | `db.createObjectStore(name, options)`       |
| Delete store  | `db.deleteObjectStore(name)`                |
| Create index  | `store.createIndex(name, keyPath, options)` |
| Add (unique)  | `store.add(value, key?)`                    |
| Add/overwrite | `store.put(value, key?)`                    |
| Get by key    | `store.get(key)`                            |
| Get all       | `store.getAll(query?, count?)`              |
| Get keys      | `store.getAllKeys(query?, count?)`          |
| Delete by key | `store.delete(key)`                         |
| Clear all     | `store.clear()`                             |
| Count         | `store.count(query?)`                       |
| Cursor        | `store.openCursor(query?, direction?)`      |
| Key cursor    | `store.openKeyCursor(query?, direction?)`   |

## Summary

1. **Open DB** with `indexedDB.open(name, version)`
2. **Create stores** in `upgradeneeded` handler
3. **Transactions** for all operations (`readonly` or `readwrite`)
4. **CRUD** via store methods (`add`, `put`, `get`, `delete`, `clear`)
5. **Search** by key ranges or via indexes
6. **Cursors** for memory-efficient iteration
7. **Promise wrapper** (idb) for async/await syntax
8. **Handle errors** and parallel version updates
