# JavaScript Animations Summary

JavaScript animations provide flexibility beyond CSS capabilities, enabling complex paths, custom timing functions, and canvas animations.

## Using setInterval (Basic)

```javascript
let start = Date.now();
let timer = setInterval(function () {
  let timePassed = Date.now() - start;

  if (timePassed >= 2000) {
    clearInterval(timer);
    return;
  }

  draw(timePassed);
}, 20);

function draw(timePassed) {
  element.style.left = timePassed / 5 + "px";
}
```

**Issues:** Multiple animations cause independent redraws, inefficient, continues when tab hidden.

## Using requestAnimationFrame (Recommended)

Schedules callbacks to run when browser is ready to repaint.

```javascript
let requestId = requestAnimationFrame(callback);
cancelAnimationFrame(requestId);
```

```javascript
let start = performance.now();

function animate(time) {
  let timePassed = time - start;

  if (timePassed < 2000) {
    draw(timePassed);
    requestAnimationFrame(animate);
  }
}

requestAnimationFrame(animate);
```

**Benefits:**

- Groups multiple animations into one repaint
- Suspends when tab is hidden (saves resources)
- Optimized for 60fps

## Universal Animation Function

```javascript
function animate({ timing, draw, duration }) {
  let start = performance.now();

  requestAnimationFrame(function animateFrame(time) {
    let timeFraction = (time - start) / duration;
    if (timeFraction > 1) timeFraction = 1;

    let progress = timing(timeFraction);
    draw(progress);

    if (timeFraction < 1) {
      requestAnimationFrame(animateFrame);
    }
  });
}
```

### Parameters

| Parameter  | Description                                  |
| ---------- | -------------------------------------------- |
| `duration` | Total animation time (ms)                    |
| `timing`   | Function mapping time (0-1) → progress (0-1) |
| `draw`     | Function that renders current progress       |

### Usage Example

```javascript
animate({
  duration: 1000,
  timing: (timeFraction) => timeFraction,
  draw(progress) {
    element.style.width = progress * 100 + "%";
  },
});
```

## Timing Functions

### Linear

```javascript
function linear(timeFraction) {
  return timeFraction;
}
```

### Power of n (Speed up)

```javascript
function quad(timeFraction) {
  return Math.pow(timeFraction, 2); // Quadratic
}

function cubic(timeFraction) {
  return Math.pow(timeFraction, 3); // Cubic
}
```

### Arc

```javascript
function circ(timeFraction) {
  return 1 - Math.sin(Math.acos(timeFraction));
}
```

### Back (Bow Shooting)

```javascript
function back(x, timeFraction) {
  return Math.pow(timeFraction, 2) * ((x + 1) * timeFraction - x);
}
// Usage: back(1.5, timeFraction)
```

### Bounce

```javascript
function bounce(timeFraction) {
  for (let a = 0, b = 1; 1; a += b, b /= 2) {
    if (timeFraction >= (7 - 4 * a) / 11) {
      return (
        -Math.pow((11 - 6 * a - 11 * timeFraction) / 4, 2) + Math.pow(b, 2)
      );
    }
  }
}
```

### Elastic

```javascript
function elastic(x, timeFraction) {
  return (
    Math.pow(2, 10 * (timeFraction - 1)) *
    Math.cos(((20 * Math.PI * x) / 3) * timeFraction)
  );
}
```

## Timing Transformations

### easeOut (Reverse)

```javascript
function makeEaseOut(timing) {
  return function (timeFraction) {
    return 1 - timing(1 - timeFraction);
  };
}

// Usage
let bounceEaseOut = makeEaseOut(bounce);
```

### easeInOut (Both Ends)

```javascript
function makeEaseInOut(timing) {
  return function (timeFraction) {
    if (timeFraction < 0.5) {
      return timing(2 * timeFraction) / 2;
    } else {
      return (2 - timing(2 * (1 - timeFraction))) / 2;
    }
  };
}

// Usage
let bounceEaseInOut = makeEaseInOut(bounce);
```

## Complete Examples

### Bouncing Ball

```javascript
function bounceBall() {
  let ball = document.getElementById("ball");
  let start = performance.now();

  function animate(time) {
    let timePassed = time - start;

    if (timePassed < 2000) {
      let progress = timePassed / 2000;

      // Bounce effect: y = sin(π * progress) * 100
      let y = Math.sin(Math.PI * progress) * 100;
      ball.style.transform = `translateY(${-y}px)`;

      requestAnimationFrame(animate);
    } else {
      ball.style.transform = "translateY(0)";
    }
  }

  requestAnimationFrame(animate);
}
```

### Moving Along Complex Path

```javascript
function moveAlongPath() {
  let element = document.getElementById("element");
  let start = performance.now();

  function timing(timeFraction) {
    // Circular path
    let angle = timeFraction * Math.PI * 2;
    return {
      x: 100 + Math.cos(angle) * 100,
      y: 100 + Math.sin(angle) * 100,
    };
  }

  function animate(time) {
    let timeFraction = Math.min(1, (time - start) / 3000);
    let position = timing(timeFraction);

    element.style.left = position.x + "px";
    element.style.top = position.y + "px";

    if (timeFraction < 1) {
      requestAnimationFrame(animate);
    }
  }

  requestAnimationFrame(animate);
}
```

### Text Typing Animation

```javascript
function typeText(text, element) {
  let index = 0;
  let start = performance.now();
  let duration = text.length * 100;

  function animate(time) {
    let timeFraction = Math.min(1, (time - start) / duration);
    let newIndex = Math.floor(timeFraction * text.length);

    if (newIndex > index) {
      index = newIndex;
      element.textContent = text.slice(0, index);
    }

    if (timeFraction < 1) {
      requestAnimationFrame(animate);
    }
  }

  requestAnimationFrame(animate);
}
```

### Bouncing Ball to the Right

```javascript
function bounceRight() {
  let ball = document.getElementById("ball");
  let start = performance.now();

  function timing(timeFraction) {
    // Bounce: sin(π * timeFraction * 4) decays
    let bounces = 4;
    let y =
      Math.sin(Math.PI * timeFraction * bounces) * (1 - timeFraction) * 50;
    return {
      x: timeFraction * 300,
      y: -y,
    };
  }

  function animate(time) {
    let timeFraction = Math.min(1, (time - start) / 2000);
    let pos = timing(timeFraction);

    ball.style.transform = `translate(${pos.x}px, ${pos.y}px)`;

    if (timeFraction < 1) {
      requestAnimationFrame(animate);
    }
  }

  requestAnimationFrame(animate);
}
```

## Performance Comparison

| Method                    | Pros                          | Cons                            |
| ------------------------- | ----------------------------- | ------------------------------- |
| **setInterval**           | Simple                        | Inefficient, runs in background |
| **setTimeout recursive**  | More control                  | Still inefficient               |
| **requestAnimationFrame** | Optimized, pauses when hidden | Requires recursive calls        |

## Summary

1. **Use `requestAnimationFrame`** for all JavaScript animations
2. **Universal pattern**: track start time, calculate elapsed, update position
3. **Timing functions** transform time (0-1) to progress (0-1)
4. **Transformations** (`easeOut`, `easeInOut`) reverse or combine timing
5. **Custom `draw`** can animate any property, canvas, or create elements
6. **Performance**: `requestAnimationFrame` groups animations, stops when tab hidden
