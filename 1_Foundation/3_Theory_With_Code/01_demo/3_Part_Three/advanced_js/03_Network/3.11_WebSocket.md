# WebSocket Summary

WebSocket provides persistent, full-duplex communication between browser and server over a single connection. Unlike HTTP, it allows real-time data exchange in both directions without repeated requests.

## Basic Connection

```javascript
// Create connection (ws:// or wss://)
let socket = new WebSocket("wss://javascript.info/chat");

// Connection opened
socket.onopen = function (e) {
  console.log("Connected");
  socket.send("Hello server!");
};

// Message received
socket.onmessage = function (event) {
  console.log("Received:", event.data);
};

// Connection closed
socket.onclose = function (event) {
  if (event.wasClean) {
    console.log(`Closed cleanly, code=${event.code}, reason=${event.reason}`);
  } else {
    console.log("Connection died");
  }
};

// Error occurred
socket.onerror = function (error) {
  console.log("Error:", error);
};
```

## Always Prefer wss://

- **wss://** is WebSocket over TLS (encrypted)
- More reliable - proxies can't inspect or block encrypted traffic
- Same as HTTPS vs HTTP

## Connection Handshake

Browser automatically sends upgrade request:

```
GET /chat
Host: javascript.info
Origin: https://javascript.info
Connection: Upgrade
Upgrade: websocket
Sec-WebSocket-Key: Iv8io/9s+lYFgZWcXczP8Q==
Sec-WebSocket-Version: 13
```

Server responds with:

```
101 Switching Protocols
Upgrade: websocket
Connection: Upgrade
Sec-WebSocket-Accept: hsBlbuDTkk24srzEOTBUlZAlC2g=
```

## Sending Data

```javascript
// Text data
socket.send("Hello");

// Binary data
let buffer = new Uint8Array([72, 101, 108, 108, 111]);
socket.send(buffer);

// Blob
let blob = new Blob(["Hello"], { type: "text/plain" });
socket.send(blob);
```

## Receiving Binary Data

```javascript
// Default: binary data as Blob
socket.onmessage = function (event) {
  if (event.data instanceof Blob) {
    console.log("Blob size:", event.data.size);
  }
};

// Switch to ArrayBuffer for binary processing
socket.binaryType = "arraybuffer";
socket.onmessage = function (event) {
  if (event.data instanceof ArrayBuffer) {
    let bytes = new Uint8Array(event.data);
    console.log(bytes);
  }
};
```

## Rate Limiting

```javascript
// Check buffered data before sending more
setInterval(() => {
  if (socket.bufferedAmount === 0) {
    socket.send(moreData());
  }
}, 100);
```

## Closing Connection

```javascript
// Close with code and reason
socket.close(1000, "Work complete");

// Receive close event
socket.onclose = function (event) {
  console.log(event.code); // 1000
  console.log(event.reason); // "Work complete"
  console.log(event.wasClean); // true
};
```

### Common Close Codes

| Code | Meaning                                             |
| ---- | --------------------------------------------------- |
| 1000 | Normal closure (default)                            |
| 1001 | Party going away (server shutdown, page navigation) |
| 1006 | Connection lost (cannot be set manually)            |
| 1009 | Message too large                                   |
| 1011 | Unexpected server error                             |

## Connection States

```javascript
console.log(socket.readyState);
// 0 = CONNECTING
// 1 = OPEN
// 2 = CLOSING
// 3 = CLOSED
```

## Subprotocols

Specify protocols the client supports:

```javascript
let socket = new WebSocket("wss://javascript.info/chat", ["soap", "wamp"]);
```

Server responds with chosen protocol:

```
Sec-WebSocket-Protocol: soap
```

## Complete Chat Example

### Client (HTML + JS)

```html
<form name="publish">
  <input type="text" name="message" />
  <input type="submit" value="Send" />
</form>
<div id="messages"></div>

<script>
  let socket = new WebSocket("wss://example.com/chat");

  // Send message from form
  document.forms.publish.onsubmit = function () {
    let message = this.message.value;
    if (message) {
      socket.send(message);
      this.message.value = "";
    }
    return false;
  };

  // Receive and display messages
  socket.onmessage = function (event) {
    let messageElem = document.createElement("div");
    messageElem.textContent = event.data;
    document.getElementById("messages").prepend(messageElem);
  };

  socket.onopen = function () {
    console.log("Connected to chat");
  };

  socket.onclose = function () {
    console.log("Disconnected");
  };
</script>
```

### Server (Node.js with ws module)

```javascript
const WebSocket = require("ws");
const wss = new WebSocket.Server({ port: 8080 });
const clients = new Set();

wss.on("connection", function (ws) {
  clients.add(ws);

  ws.on("message", function (message) {
    // Broadcast to all clients
    for (let client of clients) {
      if (client.readyState === WebSocket.OPEN) {
        client.send(message);
      }
    }
  });

  ws.on("close", function () {
    clients.delete(ws);
  });
});
```

## Reconnection Pattern

```javascript
function connect() {
  let socket = new WebSocket("wss://example.com/chat");

  socket.onopen = function () {
    console.log("Connected");
  };

  socket.onclose = function () {
    console.log("Reconnecting in 1 second...");
    setTimeout(connect, 1000);
  };

  socket.onerror = function (err) {
    console.log("Error:", err);
    socket.close(); // Trigger reconnect
  };

  return socket;
}

let socket = connect();
```

## WebSocket vs Alternatives

| Feature         | WebSocket     | Long Polling       | Server-Sent Events |
| --------------- | ------------- | ------------------ | ------------------ |
| Direction       | Bidirectional | Bidirectional      | Server→Client only |
| Overhead        | Low (once)    | High (per message) | Low                |
| Real-time       | Yes           | Delayed            | Yes                |
| Binary data     | Yes           | Via encoding       | Text only          |
| Auto-reconnect  | Manual        | Manual             | Built-in           |
| Browser support | Modern        | All                | Modern             |

## Limitations

- No built-in reconnection - must implement manually
- No authentication mechanism - handle via cookies or custom headers
- Old proxies may block ws:// (use wss:// instead)
- Server must support WebSocket protocol

## Summary

1. Create: `new WebSocket(url, [protocols])`
2. Events: `open`, `message`, `error`, `close`
3. Send: `socket.send(data)` (string, Blob, ArrayBuffer)
4. Receive: `event.data` (string or binary based on `binaryType`)
5. Close: `socket.close([code], [reason])`
6. State: `socket.readyState` (0-3)
7. Buffer check: `socket.bufferedAmount` for rate limiting
