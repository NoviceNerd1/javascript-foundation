# XMLHttpRequest Summary

`XMLHttpRequest` (XHR) is the traditional way to make HTTP requests in JavaScript. While modern code typically uses `fetch()`, XHR is still used for:

- Supporting legacy code
- Old browser compatibility
- Upload progress tracking (not available in `fetch`)

## Basic GET Request

```javascript
let xhr = new XMLHttpRequest();

// Configure request
xhr.open("GET", "/article/xmlhttprequest/example/load");

// Send request
xhr.send();

// Handle response
xhr.onload = function () {
  if (xhr.status != 200) {
    console.log(`Error ${xhr.status}: ${xhr.statusText}`);
  } else {
    console.log(`Done, got ${xhr.response.length} bytes`);
  }
};

// Track download progress
xhr.onprogress = function (event) {
  if (event.lengthComputable) {
    console.log(`Received ${event.loaded} of ${event.total} bytes`);
  }
};

// Handle network errors
xhr.onerror = function () {
  console.log("Request failed");
};
```

## Key Properties

| Property       | Description                             |
| -------------- | --------------------------------------- |
| `status`       | HTTP status code (200, 404, etc.)       |
| `statusText`   | HTTP status message ("OK", "Not Found") |
| `response`     | Server response body                    |
| `responseText` | Response as string (legacy)             |
| `responseXML`  | Response as XML document (legacy)       |
| `readyState`   | Current state (0-4)                     |
| `timeout`      | Request timeout in milliseconds         |

## Response Types

```javascript
// Set response format before send()
xhr.responseType = "json"; // '', 'text', 'arraybuffer', 'blob', 'document', 'json'

xhr.open("GET", "/api/data");
xhr.send();

xhr.onload = function () {
  let data = xhr.response; // Already parsed as JSON
  console.log(data);
};
```

## Request Headers

```javascript
// Set headers (must be after open, before send)
xhr.setRequestHeader("Content-Type", "application/json");
xhr.setRequestHeader("X-Auth", "123");

// Headers accumulate, not overwrite
xhr.setRequestHeader("X-Auth", "456");
// Result: X-Auth: 123, 456

// Get response headers
let contentType = xhr.getResponseHeader("Content-Type");
let allHeaders = xhr.getAllResponseHeaders();
```

## POST Requests

### With FormData (multipart/form-data)

```javascript
let formData = new FormData();
formData.append("name", "John");
formData.append("surname", "Smith");

let xhr = new XMLHttpRequest();
xhr.open("POST", "/api/user");
xhr.send(formData);

xhr.onload = () => console.log(xhr.response);
```

### With JSON

```javascript
let xhr = new XMLHttpRequest();
xhr.open("POST", "/api/user");
xhr.setRequestHeader("Content-Type", "application/json; charset=utf-8");

let json = JSON.stringify({ name: "John", surname: "Smith" });
xhr.send(json);
```

## Upload Progress Tracking

`xhr.upload` object provides upload progress events (not available in `fetch`):

```javascript
<input type="file" onchange="uploadFile(this.files[0])">

<script>
function uploadFile(file) {
  let xhr = new XMLHttpRequest();

  // Track upload progress
  xhr.upload.onprogress = function(event) {
    let percent = (event.loaded / event.total * 100).toFixed(1);
    console.log(`Uploaded ${percent}%`);
  };

  xhr.upload.onload = function() {
    console.log('Upload complete');
  };

  xhr.upload.onerror = function() {
    console.log('Upload failed');
  };

  xhr.onload = function() {
    console.log('Server response:', xhr.response);
  };

  xhr.open('POST', '/upload');
  xhr.send(file);
}
</script>
```

## Ready States

```javascript
// xhr.readyState values:
// 0 - UNSENT - initial state
// 1 - OPENED - open() called
// 2 - HEADERS_RECEIVED - response headers received
// 3 - LOADING - downloading response
// 4 - DONE - request complete

xhr.onreadystatechange = function () {
  if (xhr.readyState === 3) {
    console.log("Loading...");
  }
  if (xhr.readyState === 4) {
    console.log("Complete!");
  }
};
```

## Abort Request

```javascript
let xhr = new XMLHttpRequest();
xhr.open("GET", "/large-file");
xhr.send();

// Later, cancel request
xhr.abort();

xhr.onabort = function () {
  console.log("Request aborted");
};
```

## Timeout

```javascript
let xhr = new XMLHttpRequest();
xhr.timeout = 10000; // 10 seconds
xhr.open("GET", "/slow-endpoint");
xhr.send();

xhr.ontimeout = function () {
  console.log("Request timed out");
};
```

## Cross-Origin Requests (CORS)

```javascript
let xhr = new XMLHttpRequest();
xhr.withCredentials = true; // Send cookies to other origin
xhr.open("GET", "https://api.another.com/data");
xhr.send();

xhr.onload = function () {
  console.log(xhr.response);
};
```

## Complete File Upload Example

```html
<input type="file" id="fileInput" />
<progress id="progressBar" value="0" max="100"></progress>
<button onclick="upload()">Upload</button>

<script>
  function upload() {
    let file = document.getElementById("fileInput").files[0];
    let progress = document.getElementById("progressBar");
    let xhr = new XMLHttpRequest();

    // Upload progress
    xhr.upload.onprogress = function (e) {
      if (e.lengthComputable) {
        let percent = (e.loaded / e.total) * 100;
        progress.value = percent;
      }
    };

    xhr.onload = function () {
      if (xhr.status === 200) {
        console.log("Upload complete:", xhr.response);
      } else {
        console.error("Upload failed:", xhr.status);
      }
    };

    xhr.onerror = function () {
      console.error("Network error");
    };

    xhr.open("POST", "/upload");
    xhr.send(file);
  }
</script>
```

## XMLHttpRequest vs fetch

| Feature           | XMLHttpRequest        | fetch               |
| ----------------- | --------------------- | ------------------- |
| Upload progress   | ✅ Yes (`xhr.upload`) | ❌ No               |
| Download progress | ✅ Yes (onprogress)   | ✅ Yes (via stream) |
| Promise-based     | ❌ No (events)        | ✅ Yes              |
| Syntax            | Verbose               | Clean               |
| Abort             | `xhr.abort()`         | AbortController     |
| Response types    | responseType property | Response methods    |
| Browser support   | All browsers          | Modern browsers     |

## Summary

1. **Create**: `new XMLHttpRequest()`
2. **Configure**: `xhr.open(method, url)`
3. **Set headers**: `xhr.setRequestHeader()`
4. **Send**: `xhr.send(body)`
5. **Handle**: `onload`, `onerror`, `onprogress`
6. **Upload tracking**: Use `xhr.upload` events
7. **Response**: Access via `xhr.response`
