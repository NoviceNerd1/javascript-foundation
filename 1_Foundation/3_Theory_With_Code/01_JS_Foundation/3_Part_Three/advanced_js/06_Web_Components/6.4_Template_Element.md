# Template Element Summary

The `<template>` element stores HTML markup that the browser ignores until inserted into the document. It's ideal for reusable component structures.

## Key Features

| Feature               | Description                                                      |
| --------------------- | ---------------------------------------------------------------- |
| **Content ignored**   | Styles not applied, scripts not executed, no rendering           |
| **Syntax validation** | Browser validates HTML syntax inside                             |
| **Any HTML allowed**  | Can contain rows, cells, or any elements without proper wrappers |
| **Clonable**          | `template.content.cloneNode(true)` creates reusable copies       |

## Basic Usage

```html
<template id="tmpl">
  <style>
    .message {
      font-weight: bold;
      color: blue;
    }
  </style>
  <div class="message">Hello, world!</div>
  <script>
    console.log("This runs when inserted");
  </script>
</template>

<script>
  // Clone and insert
  let div = document.createElement("div");
  div.append(tmpl.content.cloneNode(true));
  document.body.append(div);
  // Now script runs and styles apply
</script>
```

## Template Content Properties

```javascript
let template = document.getElementById("tmpl");

// Access as DocumentFragment
let fragment = template.content;
console.log(fragment.nodeType); // 11 (DocumentFragment)

// Clone before inserting
let clone = template.content.cloneNode(true);
document.body.appendChild(clone);
```

## Why Not Just Hidden HTML?

```html
<!-- Invalid: browser adds <tbody> wrapper -->
<div style="display: none">
  <tr>
    <td>Contents</td>
  </tr>
</div>

<!-- Valid: <template> preserves exact structure -->
<template>
  <tr>
    <td>Contents</td>
  </tr>
</template>
```

## Shadow DOM with Template

```html
<template id="tooltip-tmpl">
  <style>
    :host {
      position: relative;
      display: inline-block;
    }
    .tooltip-text {
      position: absolute;
      background: #333;
      color: white;
      padding: 5px;
      border-radius: 3px;
      display: none;
    }
    :host(:hover) .tooltip-text {
      display: block;
    }
  </style>
  <slot></slot>
  <span class="tooltip-text">
    <slot name="tooltip">Default tooltip</slot>
  </span>
</template>

<script>
  class Tooltip extends HTMLElement {
    constructor() {
      super();
      this.attachShadow({ mode: "open" });
      const template = document.getElementById("tooltip-tmpl");
      this.shadowRoot.append(template.content.cloneNode(true));
    }
  }

  customElements.define("custom-tooltip", Tooltip);
</script>

<custom-tooltip>
  Hover me
  <span slot="tooltip">Custom tooltip text</span>
</custom-tooltip>
```

## Multiple Clone Example

```javascript
const template = document.getElementById("item-tmpl");
const list = document.getElementById("list");

for (let i = 0; i < 10; i++) {
  let clone = template.content.cloneNode(true);
  clone.querySelector(".title").textContent = `Item ${i}`;
  list.appendChild(clone);
}
```

## Template with Slots

```html
<template id="card-tmpl">
  <style>
    .card {
      border: 1px solid #ccc;
      border-radius: 8px;
      padding: 16px;
      margin: 8px;
    }
    .card h3 {
      margin: 0 0 8px 0;
    }
  </style>
  <div class="card">
    <h3><slot name="title">Default Title</slot></h3>
    <div><slot name="content">Default Content</slot></div>
  </div>
</template>

<div id="cards"></div>

<script>
  const template = document.getElementById("card-tmpl");
  const container = document.getElementById("cards");

  function addCard(title, content) {
    const clone = template.content.cloneNode(true);
    clone.querySelector('[slot="title"]').textContent = title;
    clone.querySelector('[slot="content"]').textContent = content;
    container.appendChild(clone);
  }

  addCard("Hello", "World!");
  addCard("Template", "With slots");
</script>
```

## Dynamic Template Loading

```html
<template id="loading-tmpl">
  <div class="loading-spinner">
    <div class="spinner"></div>
    <span>Loading...</span>
  </div>
  <style>
    .loading-spinner {
      display: flex;
      align-items: center;
      gap: 8px;
      padding: 16px;
    }
    .spinner {
      width: 20px;
      height: 20px;
      border: 2px solid #ccc;
      border-top-color: #333;
      border-radius: 50%;
      animation: spin 0.6s linear infinite;
    }
    @keyframes spin {
      to {
        transform: rotate(360deg);
      }
    }
  </style>
</template>

<script>
  function showLoading(container) {
    const template = document.getElementById("loading-tmpl");
    const loadingElem = template.content.cloneNode(true);
    container.appendChild(loadingElem);
    return loadingElem;
  }

  function hideLoading(loadingElem) {
    loadingElem.remove();
  }

  // Usage
  let container = document.getElementById("results");
  let loading = showLoading(container);
  fetch("/api/data").finally(() => hideLoading(loading));
</script>
```

## Template with Data Attributes

```html
<template id="user-tmpl">
  <div class="user-card">
    <img class="avatar" />
    <div class="info">
      <h3 class="name"></h3>
      <p class="email"></p>
    </div>
  </div>
</template>

<script>
  class UserCard {
    constructor(userData) {
      const template = document.getElementById("user-tmpl");
      this.element = template.content.cloneNode(true);

      this.element.querySelector(".avatar").src = userData.avatar;
      this.element.querySelector(".name").textContent = userData.name;
      this.element.querySelector(".email").textContent = userData.email;
    }

    mount(container) {
      container.appendChild(this.element);
    }
  }

  const user = new UserCard({
    avatar: "john.jpg",
    name: "John Doe",
    email: "john@example.com",
  });
  user.mount(document.getElementById("users"));
</script>
```

## Browser Support

- Supported in all modern browsers
- IE not supported (can be polyfilled)

## Summary

1. **`<template>`** stores inert HTML markup
2. **Content accessible** via `template.content` (DocumentFragment)
3. **Clone before use**: `template.content.cloneNode(true)`
4. **Any HTML allowed**: no browser "fixing" of structure
5. **Becomes live** when inserted into document (styles apply, scripts run)
6. **Perfect for** reusable components, Shadow DOM templates, dynamic content
7. **No built-in data binding**: implement manually or use with frameworks
