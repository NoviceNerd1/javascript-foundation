# Resumable File Upload Summary

Resumable uploads allow file uploads to continue after connection loss. Since `fetch()` doesn't track upload progress, we use `XMLHttpRequest` with server coordination.

## Core Concepts

1. **Unique file ID** - Identifies the file being uploaded
2. **Server tracking** - Server stores how many bytes were received
3. **Resume point** - Client asks server for current progress, then uploads remaining bytes

## Implementation Steps

### Step 1: Create Unique File ID

```javascript
let fileId = file.name + "-" + file.size + "-" + file.lastModified;
```

Combines name, size, and lastModified to create unique identifier. If any property changes, the file is considered different.

### Step 2: Query Server for Progress

```javascript
async function getUploadedBytes(fileId) {
  let response = await fetch("/status", {
    headers: {
      "X-File-Id": fileId,
    },
  });
  return await response.text(); // Returns number of bytes already uploaded
}
```

### Step 3: Upload Remaining Portion

```javascript
async function uploadFile(file, startByte) {
  let xhr = new XMLHttpRequest();
  xhr.open("POST", "/upload");

  // Tell server which file and where to resume
  xhr.setRequestHeader("X-File-Id", fileId);
  xhr.setRequestHeader("X-Start-Byte", startByte);

  // Track upload progress
  xhr.upload.onprogress = (e) => {
    let uploaded = startByte + e.loaded;
    let total = startByte + e.total;
    console.log(
      `Uploaded ${uploaded} of ${total} bytes (${Math.round((uploaded / total) * 100)}%)`,
    );
  };

  xhr.onload = () => {
    if (xhr.status === 200) {
      console.log("Upload complete");
    } else {
      console.log("Upload failed:", xhr.status);
    }
  };

  xhr.onerror = () => console.log("Network error");

  // Send only the remaining part
  let remainingBlob = file.slice(startByte);
  xhr.send(remainingBlob);
}
```

### Step 4: Complete Resumable Upload Function

```javascript
async function resumableUpload(file) {
  let fileId = file.name + "-" + file.size + "-" + file.lastModified;

  try {
    // Check how much server already has
    let startByte = await getUploadedBytes(fileId);

    if (startByte === 0) {
      console.log("Starting new upload");
    } else if (startByte === file.size) {
      console.log("File already uploaded");
      return;
    } else {
      console.log(`Resuming from byte ${startByte}`);
    }

    // Upload remaining bytes
    await uploadFile(file, startByte);
  } catch (err) {
    console.log("Upload interrupted, can resume later");
  }
}
```

## Server-Side Implementation (Node.js Example)

```javascript
const fs = require("fs");
const express = require("express");
const app = express();

// Store upload progress
const uploads = new Map(); // fileId -> bytes uploaded

// Status endpoint
app.get("/status", (req, res) => {
  let fileId = req.headers["x-file-id"];
  let uploaded = uploads.get(fileId) || 0;
  res.send(uploaded.toString());
});

// Upload endpoint
app.post("/upload", (req, res) => {
  let fileId = req.headers["x-file-id"];
  let startByte = parseInt(req.headers["x-start-byte"]) || 0;

  let filePath = `./uploads/${fileId}`;
  let writeStream = fs.createWriteStream(filePath, {
    flags: startByte === 0 ? "w" : "a",
  });

  req.on("data", (chunk) => {
    writeStream.write(chunk);
    // Update progress
    let current = uploads.get(fileId) || 0;
    uploads.set(fileId, current + chunk.length);
  });

  req.on("end", () => {
    writeStream.end();
    res.send("ok");
  });

  req.on("error", (err) => {
    console.error("Upload error:", err);
    res.status(500).send("error");
  });
});
```

## Complete Client Example with Retry

```javascript
class ResumableUploader {
  constructor(url, statusUrl) {
    this.uploadUrl = url;
    this.statusUrl = statusUrl;
    this.retryDelay = 3000; // 3 seconds
    this.maxRetries = 5;
  }

  async upload(file, onProgress) {
    let fileId = file.name + "-" + file.size + "-" + file.lastModified;
    let startByte = await this.getProgress(fileId);

    if (startByte === file.size) {
      console.log("Already uploaded");
      return;
    }

    return this.sendChunk(file, fileId, startByte, onProgress);
  }

  async getProgress(fileId) {
    try {
      let response = await fetch(this.statusUrl, {
        headers: { "X-File-Id": fileId },
      });
      return parseInt(await response.text());
    } catch {
      return 0; // Assume nothing uploaded
    }
  }

  async sendChunk(file, fileId, startByte, onProgress, retryCount = 0) {
    return new Promise((resolve, reject) => {
      let xhr = new XMLHttpRequest();
      xhr.open("POST", this.uploadUrl);

      xhr.setRequestHeader("X-File-Id", fileId);
      xhr.setRequestHeader("X-Start-Byte", startByte);

      xhr.upload.onprogress = (e) => {
        if (onProgress) {
          onProgress(startByte + e.loaded, startByte + e.total);
        }
      };

      xhr.onload = () => {
        if (xhr.status === 200) {
          resolve();
        } else {
          reject(new Error(`HTTP ${xhr.status}`));
        }
      };

      xhr.onerror = () => reject(new Error("Network error"));
      xhr.ontimeout = () => reject(new Error("Timeout"));

      xhr.send(file.slice(startByte));
    }).catch(async (err) => {
      if (retryCount < this.maxRetries) {
        console.log(`Retry ${retryCount + 1}...`);
        await new Promise((r) => setTimeout(r, this.retryDelay));

        // Get latest progress before retry
        let newStartByte = await this.getProgress(fileId);
        return this.sendChunk(
          file,
          fileId,
          newStartByte,
          onProgress,
          retryCount + 1,
        );
      }
      throw err;
    });
  }
}

// Usage
let uploader = new ResumableUploader("/upload", "/status");
let file = document.getElementById("fileInput").files[0];

uploader
  .upload(file, (uploaded, total) => {
    let percent = ((uploaded / total) * 100).toFixed(1);
    console.log(`Progress: ${percent}% (${uploaded}/${total} bytes)`);
  })
  .then(() => {
    console.log("Upload complete!");
  })
  .catch((err) => {
    console.log("Upload failed:", err);
  });
```

## Limitations & Considerations

| Aspect                 | Solution                            |
| ---------------------- | ----------------------------------- |
| **File ID uniqueness** | Use name + size + lastModified      |
| **Server storage**     | Track uploaded bytes per fileId     |
| **Concurrent uploads** | File locking or atomic writes       |
| **Large files**        | Use `Blob.slice()` to send chunks   |
| **Network failure**    | Implement retry with progress query |
| **Browser support**    | XMLHttpRequest, Blob.slice()        |

## Why XMLHttpRequest Instead of fetch?

- `fetch()` cannot track upload progress
- `xhr.upload.onprogress` provides upload progress events
- `XMLHttpRequest` supports `timeout` and `abort` for better control

## Summary

1. Generate unique file ID from file metadata
2. Query server for already uploaded bytes
3. Send remaining bytes using `Blob.slice()`
4. Track upload progress with `xhr.upload.onprogress`
5. Handle failures with retry mechanism
6. Server must track uploaded bytes per file ID
7. Use `X-Start-Byte` header to indicate resume position
